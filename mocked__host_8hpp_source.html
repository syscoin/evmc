150  void record_account_access(const address& addr) const
151  {
152  if (recorded_account_accesses.empty())
153  recorded_account_accesses.reserve(max_recorded_account_accesses);
154 
155  if (recorded_account_accesses.size() < max_recorded_account_accesses)
156  recorded_account_accesses.emplace_back(addr);

159 public:
161  bool account_exists(const address& addr) const noexcept override
162  {
163  record_account_access(addr);
164  return accounts.count(addr) != 0;
165  }

168  bytes32 get_storage(const address& addr, const bytes32& key) const noexcept override
169  {
170  record_account_access(addr);
171 
172  const auto account_iter = accounts.find(addr);
173  if (account_iter == accounts.end())
174  return {};
175 
176  const auto storage_iter = account_iter->second.storage.find(key);
177  if (storage_iter != account_iter->second.storage.end())
178  return storage_iter->second.value;
179  return {};
180  }
181 
183  evmc_storage_status set_storage(const address& addr,
184  const bytes32& key,
185  const bytes32& value) noexcept override
186  {
187  record_account_access(addr);
188 
189  // Get the reference to the old value.
190  // This will create the account in case it was not present.
191  // This is convenient for unit testing and standalone EVM execution to preserve the
192  // storage values after the execution terminates.
193  auto& old = accounts[addr].storage[key];
194 
195  // Follow https://eips.ethereum.org/EIPS/eip-1283 specification.
196  // WARNING! This is not complete implementation as refund is not handled here.
197 
198  if (old.value == value)
199  return EVMC_STORAGE_UNCHANGED;
200 
201  evmc_storage_status status{};
202  if (!old.dirty)
203  {
204  old.dirty = true;
205  if (!old.value)
206  status = EVMC_STORAGE_ADDED;
207  else if (value)
208  status = EVMC_STORAGE_MODIFIED;
209  else
210  status = EVMC_STORAGE_DELETED;
211  }
212  else
213  status = EVMC_STORAGE_MODIFIED_AGAIN;
214 
215  old.value = value;
216  return status;
217  }
218

251  size_t copy_code(const address& addr,
252  size_t code_offset,
253  uint8_t* buffer_data,
254  size_t buffer_size) const noexcept override
255  {
256  record_account_access(addr);
257  const auto it = accounts.find(addr);
258  if (it == accounts.end())
259  return 0;
260 
261  const auto& code = it->second.code;
262 
263  if (code_offset >= code.size())
264  return 0;
265 
266  const auto n = std::min(buffer_size, code.size() - code_offset);
267 
268  if (n > 0)
269  std::copy_n(&code[code_offset], n, buffer_data);
270  return n;
271  }

274  void selfdestruct(const address& addr, const address& beneficiary) noexcept override
275  {
276  record_account_access(addr);
277  recorded_selfdestructs.push_back({addr, beneficiary});
278  }

281  result call(const evmc_message& msg) noexcept override
282  {
283  record_account_access(msg.destination);
284 
285  if (recorded_calls.empty())
286  {
287  recorded_calls.reserve(max_recorded_calls);
288  m_recorded_calls_inputs.reserve(max_recorded_calls); // Iterators will not invalidate.
289  }
290 
291  if (recorded_calls.size() < max_recorded_calls)
292  {
293  recorded_calls.emplace_back(msg);
294  auto& call_msg = recorded_calls.back();
295  if (call_msg.input_size > 0)
296  {
297  m_recorded_calls_inputs.emplace_back(call_msg.input_data, call_msg.input_size);
298  const auto& input_copy = m_recorded_calls_inputs.back();
299  call_msg.input_data = input_copy.data();
300  }
301  }
302  return result{call_result};
303  }
304 
306  evmc_tx_context get_tx_context() const noexcept override { return tx_context; }
307 
309  bytes32 get_block_hash(int64_t block_number) const noexcept override
310  {

312  return block_hash;

341  evmc_access_status access_account(const address& addr) noexcept override
342  {
343  // Check if the address have been already accessed.
344  const auto already_accessed =
345  std::find(recorded_account_accesses.begin(), recorded_account_accesses.end(), addr) !=
346  recorded_account_accesses.end();
347 
348  record_account_access(addr);
349 
350  // Accessing precompiled contracts is always warm.
351  if (addr >= 0x0000000000000000000000000000000000000001_address &&
352  addr <= 0x0000000000000000000000000000000000000009_address)
353  return EVMC_ACCESS_WARM;
354 
355  return already_accessed ? EVMC_ACCESS_WARM : EVMC_ACCESS_COLD;
356  }
357 
371  evmc_access_status access_storage(const address& addr, const bytes32& key) noexcept override
372  {
373  auto& value = accounts[addr].storage[key];
374  const auto access_status = value.access_status;
375  value.access_status = EVMC_ACCESS_WARM;
376  return access_status;
377  }
